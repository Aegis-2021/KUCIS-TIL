- A&&B 문법 = A가 참일 때 B가 실행됨.

```tsx
import React, { useState } from "react";
import { IWord } from "./Interface";

const Word = ({ word }: IWord) => {
  const [isShow, setIsShow] = useState<boolean>(false);
  return (
    <>
      <tr key={word.id}>
        <td>
          <input type="checkbox" />
        </td>
        <td>{word.eng}</td>
        <td>{isShow && word.kor}</td> -> isShow가 true일때 word.kor가 실행되도록 함.
        <button>뜻 보기</button>
        <button className="btn_del">삭제하기</button>
      </tr>
    </>
  );
};

export default Word;
```

  - 반복되는 로직은 커스텀 훅으로 만들어서 사용 가능!

    ```tsx
    useEffect(() => {
      fetch(`http://localhost:3001/words?day=${day}`)
        .then((res) => {
          return res.json();
        })
        .then((data) => {
          setWords(data);
        });
    }, [day]);
    ```

    →

    ```tsx
    export default function useFetch(url: any) {
      const [data, setData] = useState<Array<any>>([]);

      useEffect(() => {
        fetch(url)
          .then((res) => {
            return res.json();
          })
          .then((data) => {
            setData(data);
          });
      }, [url]);

      return data;
    }
    ```
